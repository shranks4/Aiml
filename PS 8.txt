    // Frontend validation
    function validateForm() {
      const name = document.getElementById("name").value.trim();
      const rating = parseInt(document.getElementById("rating").value);
      const feedback = document.getElementById("feedback").value.trim();


      // Prevent script tags
      if (/<script>/i.test(name) || /<script>/i.test(feedback)) {
        alert("❌ Script tags are not allowed!");
        return false;
      }


      // Name validation (letters only)
      if (!/^[A-Za-z\s]+$/.test(name)) {
        alert("❌ Name must contain only letters!");
        return false;
      }


      // Rating validation
      if (rating < 1 || rating > 5) {
        alert("❌ Rating must be between 1 and 5!");
        return false;
      }


      // Feedback length
      if (feedback.length < 10) {
        alert("❌ Feedback must be at least 10 characters long!");
        return false;
      }


      return true;
    }

server.js
// server.js
const express = require("express");
const app = express();
app.use(express.urlencoded({ extended: true }));


app.post("/submit", (req, res) => {
  const { name, email, rating, feedback } = req.body;


  // === Backend validation ===
  if (!name || !email || !rating || !feedback) {
    return res.send("❌ Please fill all fields!");
  }


  // Prevent script tags
  const scriptRegex = /<script.*?>.*?<\/script>/i;
  if (scriptRegex.test(name) || scriptRegex.test(feedback)) {
    return res.send("❌ Script tags are not allowed!");
  }


  // Name validation
  if (!/^[A-Za-z\s]+$/.test(name)) {
    return res.send("❌ Invalid name (letters only allowed).");
  }


  // Email already exists check (simulated)
  if (email === "test@gmail.com") {
    return res.send("❌ Email already exists in database.");
  }


  // Rating validation
  if (rating < 1 || rating > 5) {
    return res.send("❌ Rating must be between 1 and 5.");
  }


  // Feedback content validation
  if (feedback.length < 10) {
    return res.send("❌ Feedback is too short.");
  }


  if (/select|drop|insert|delete|update/i.test(feedback)) {
    return res.send("❌ Possible SQL injection detected.");
  }


  // Success response
  res.send(`
    ✅ Feedback submitted successfully!<br>
    <b>Name:</b> ${name}<br>
    <b>Email:</b> ${email}<br>
    <b>Rating:</b> ${rating}<br>
    <b>Feedback:</b> ${feedback}<br>
  `);
});


app.listen(3000, () => console.log("🚀 Server running at http://localhost:3000"));
